# git log
  - help: man git-log
  - git log : chronologically showing logs
  - git log -2 : only show two logs
  - git log -p : including patches
  - git log --pretty=oneline  : shows by one line format
    - try --pretty=full, --pretty=fuller
  - git log --since "2 months"
    - try --until "2 months"
  - git log --graph
